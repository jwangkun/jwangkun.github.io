K8S部署eureka集群高可用

Posted on 2020-10-21，3 min read

pipeline {
agent {
node {
label 'maven'
}
}
parameters {
string(name:'TAG_NAME',defaultValue:'latest',description:'版本号')
choice(name:'PROJECT_NAME', choices:['third-party-service','cms-service','app-service','applet-service','resource-service','gateway'], description: '需要部署项目')
}
environment {
DOCKER_CREDENTIAL_ID = ''
GOGS_CREDENTIAL_ID = ''
KUBECONFIG_CREDENTIAL_ID = ''
REGISTRY = ''
ALIBABA_DOCKER_NAMESPACE = ''
GOGS_ACCOUNT = ''
SONAR_CREDENTIAL_ID = ''
BRANCH_NAME = ''
}
stages {
stage('拉取代码') {
steps {
git(credentialsId: 'git', url: 'https://git.xxxxx.com', branch: 'master', changelog: true, poll: false)
sh 'echo 正在构建 $PROJECT_NAME 版本号 $TAG_NAME'

    container ('maven') {
      sh "mvn clean install -Dprofiles.active=pro -Dmaven.test.skip=true -gs `pwd`/mvn-settings.xml"

    }
  }
}
stage('sonarqube代码质量分析') {
  steps {
    container ('maven') {
      withCredentials([string(credentialsId: "$SONAR_CREDENTIAL_ID", variable: 'SONAR_TOKEN')]) {
        withSonarQubeEnv('sonar') {
          sh "mvn sonar:sonar -gs `pwd`/mvn-settings.xml -Dsonar.projectKey=tzy_cloud -Dsonar.host.url=http://sonarqube-sonarqube.kubesphere-devops-system:9000 -Dsonar.login=$SONAR_TOKEN"
        }
      }
      timeout(time: 1, unit: 'HOURS') {
        waitForQualityGate abortPipeline: true
      }
    }
  }
}
stage ('构建镜像-推送快照镜像') {
  steps {
    container ('maven') {
        sh 'mvn -Dprofiles.active=pro -Dmaven.test.skip=true -gs `pwd`/mvn-settings.xml clean package'
        script {
           if("$PROJECT_NAME"=='gateway'){
              sh 'echo 网关环境'
              sh 'cd $PROJECT_NAME && docker build -f Dockerfile -t $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:SNAPSHOT-$BRANCH_NAME-$BUILD_NUMBER .'
           }else{
              sh 'echo $PROJECT_NAME 环境'
              sh 'cd service/$PROJECT_NAME && docker build -f Dockerfile -t $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:SNAPSHOT-$BRANCH_NAME-$BUILD_NUMBER .'
           }
         }
        withCredentials([usernamePassword(passwordVariable : 'DOCKER_PASSWORD' ,usernameVariable : 'DOCKER_USERNAME' ,credentialsId : "$DOCKER_CREDENTIAL_ID" ,)]) {
            sh 'echo "$DOCKER_PASSWORD" | docker login $REGISTRY -u "$DOCKER_USERNAME" --password-stdin'
            sh 'docker tag  $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:SNAPSHOT-$BRANCH_NAME-$BUILD_NUMBER $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:$TAG_NAME'
            sh 'docker push  $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:$TAG_NAME'
            sh 'docker tag  $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:SNAPSHOT-$BRANCH_NAME-$BUILD_NUMBER $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:latest'
            sh 'docker push  $REGISTRY/$ALIBABA_DOCKER_NAMESPACE/$PROJECT_NAME:latest'
        }
    }
  }
}
stage('部署到k8s') {
  steps {
    input(id: 'deploy-to-dev', message: "是否将 $PROJECT_NAME 部署到k8s集群中？")
    script {
       if("$PROJECT_NAME"=='gateway'){
          kubernetesDeploy(configs: "$PROJECT_NAME/deploy/**", enableConfigSubstitution: true, kubeconfigId: "$KUBECONFIG_CREDENTIAL_ID")
       }else{
          kubernetesDeploy(configs: "service/$PROJECT_NAME/deploy/**", enableConfigSubstitution: true, kubeconfigId: "$KUBECONFIG_CREDENTIAL_ID")
       }
     }
  }
}
stage('发布版本'){
  when{
    expression{
      return params.TAG_NAME =~ /v.*/
    }
  }
  steps {
    container ('maven') {
      input(id: 'release-git-with-tag', message: '发布带tag到git?')
      withCredentials([usernamePassword(credentialsId: "$GOGS_CREDENTIAL_ID", passwordVariable: 'GIT_PASSWORD', usernameVariable: 'GIT_USERNAME')]) {
         sh 'git config --global user.email "***@qq.com"'
         sh 'git config --global user.name "****"'
         sh 'git tag -a $TAG_NAME -m "$TAG_NAME" '
         sh 'git push http://$GIT_USERNAME:$GIT_PASSWORD@git.xxxxx.com --tags --ipv4'
      }
    }
  }
}

}
}
